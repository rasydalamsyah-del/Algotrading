   248	def _classify_raw(iset: IndicatorSet) -> Tuple[MarketRegime, float]:
   249	    if not iset.trend.is_valid() or not iset.strength.is_valid():
   250	        return MarketRegime.UNDEFINED, 0.25
   251	
   252	    if _is_volatile(iset):
   253	        conf = _calc_confidence(iset, MarketRegime.VOLATILE_EXPANSION)
   254	        return MarketRegime.VOLATILE_EXPANSION, conf
   255	
   256	    adx       = iset.strength.adx
   257	    plus_di   = iset.strength.plus_di
   258	    minus_di  = iset.strength.minus_di
   259	    has_adx   = adx is not None
   260	
   261	    bullish_pairs = _count_bullish_ema_pairs(iset)
   262	    bearish_pairs = _count_bearish_ema_pairs(iset)
   263	    st_bull       = _supertrend_is_bull(iset)
   264	
   265	    trending = has_adx and adx >= REGIME_TRENDING_ADX_MIN
   266	
   267	    is_bear = (
   268	        bearish_pairs >= REGIME_BEAR_EMA_REQUIRED_PAIRS
   269	        and (st_bull is None or st_bull is False)
   270	    )
   271	    if trending and is_bear and plus_di is not None and minus_di is not None:
   272	        if minus_di > plus_di:
   273	            conf = _calc_confidence(iset, MarketRegime.TRENDING_BEAR)
   274	            return MarketRegime.TRENDING_BEAR, conf
   275	    # [SHORT-FIX F1 -- asimetri fallback DI] Bull di bawah punya fallback
   276	    # `plus_di is None -> tetap BULL x0.80`; bear SEBELUMNYA tidak (DI hilang
   277	    # = langsung jatuh UNDEFINED). Akibat terukur: 287/307 (93%) kandidat
   278	    # short regime undefined -> threshold 59-68 + modifier 0.5-0.75 ->
   279	    # NOL short pernah lolos trigger. Fallback bear ini SENGAJA lebih ketat
   280	    # dari bull: wajib konfirmasi supertrend bearish eksplisit (st_bull is
   281	    # False, bukan None) sebagai pengganti bukti DI yang hilang.
   282	    elif trending and is_bear and (plus_di is None or minus_di is None):
   283	        if st_bull is False:
   284	            conf = _calc_confidence(iset, MarketRegime.TRENDING_BEAR) * 0.80
   285	            return MarketRegime.TRENDING_BEAR, round(conf, 3)
   286	
   287	    is_bull = bullish_pairs >= REGIME_BULL_EMA_REQUIRED_PAIRS
   288	    if trending and is_bull:
   289	        if plus_di is not None and minus_di is not None and plus_di > minus_di:
   290	            conf = _calc_confidence(iset, MarketRegime.TRENDING_BULL)
   291	            return MarketRegime.TRENDING_BULL, conf
   292	        elif plus_di is None:
   293	            conf = _calc_confidence(iset, MarketRegime.TRENDING_BULL) * 0.80
   294	            return MarketRegime.TRENDING_BULL, round(conf, 3)
   295	
   296	    if _is_ranging(iset):
   297	        conf = _calc_confidence(iset, MarketRegime.RANGING)
   298	        return MarketRegime.RANGING, conf
   299	
   300	    return MarketRegime.UNDEFINED, 0.35

   346	def is_tradeable_regime(
   347	    regime: MarketRegime,
   348	    confidence: float,
   349	    allowed_regimes: Optional[list] = None,
   350	    min_confidence: float = REGIME_MIN_CONFIDENCE_TO_TRADE,
   351	    side: str = "long",
   352	    # [BIAS-FIX] Default "long" -- semua caller lama (spot, dan future
   353	    # sebelum threading side ditambahkan di titik lain) tetap dapat
   354	    # perilaku identik persis dengan sebelum parameter ini ada.
   355	) -> Tuple[bool, str]:
   356	    if confidence < min_confidence:
   357	        return False, (
   358	            f"Regime confidence terlalu rendah: {confidence:.2f} < {min_confidence:.2f}"
   359	        )
   360	
   361	    if allowed_regimes is not None:
   362	        if regime.value not in allowed_regimes:
   363	            return False, (
   364	                f"Regime '{regime.value}' tidak diizinkan. "
   365	                f"Allowed: {allowed_regimes}"
   366	            )
   367	
   368	    # [BUG-FIX] Sebelumnya hardcode "if regime == TRENDING_BEAR" di sini —
   369	    # padahal MarketRegime.allows_long property sudah ada persis untuk
   370	    # tujuan ini (dan sudah lama jadi dead code karena tak dipakai siapa pun).
   371	    # Pakai allows_long sebagai satu sumber kebenaran: otomatis ikut kalau
   372	    # daftar regime yang diizinkan long berubah di masa depan, tidak perlu
   373	    # sinkronkan manual di 2 tempat berbeda.
   374	    #
   375	    # [BIAS-FIX -- short regime gate] Sebelumnya SELALU cek allows_long
   376	    # apapun side-nya -- akibatnya regime trending_bear (ideal utk short)
   377	    # SELALU tertolak di sini bahkan utk kandidat short, karena tidak ada
   378	    # padanan allows_short yang pernah dicek. Sekarang pilih properti sesuai
   379	    # side; allows_long tetap satu-satunya sumber kebenaran utk long (tidak
   380	    # berubah), allows_short (baru) sumber kebenaran utk short.
   381	    is_long = side != "short"
   382	    regime_ok = regime.allows_long if is_long else regime.allows_short
   383	    if not regime_ok:
   384	        return False, (
   385	            f"Regime '{regime.value}' tidak mengizinkan posisi {side} "
   386	            f"(allows_{side}=False)."
   387	        )
   388	
   389	    return True, f"Regime {regime.value} OK (confidence={confidence:.2f})"

   443	                symbol, record.regime, record.regime_confidence, record.timestamp,
   444	            )

   446	        except Exception as exc:
   447	            log.warning("Gagal restore regime [%s]: %s", symbol, exc)
   448	    log.info("Regime restore selesai: %d/%d symbol.", restored, len(symbols))

   450	def is_regime_transition(symbol: str) -> tuple:
   451	    """Cek apakah regime symbol sedang dalam transisi (pending != current).
   452	    Return: (is_transitioning: bool, from_regime: str, to_regime: str)
   453	    """
   454	    with _REGIME_BUFFERS_LOCK:
   455	        buf = _get_buffer(symbol)
   456	        if buf.pending_regime == MarketRegime.UNDEFINED:
   457	            return False, buf.current_regime.value, buf.current_regime.value
   458	        return True, buf.current_regime.value, buf.pending_regime.value

   624	    def validate(self) -> List[str]:
   625	        errors: List[str] = []
   626	
   627	        if self.rsi_min >= self.rsi_max:
   628	            errors.append(
   629	                f"[{self.profile_name}] rsi_min ({self.rsi_min}) >= rsi_max ({self.rsi_max})"
   630	            )
   631	
   632	        if self.quick_sl_pct >= self.quick_tp_pct:
   633	            errors.append(
   634	                f"[{self.profile_name}] quick_sl_pct ({self.quick_sl_pct}) "
   635	                f">= quick_tp_pct ({self.quick_tp_pct})"
   636	            )
   637	
   638	        rr = self.atr_tp_mult / self.atr_sl_mult if self.atr_sl_mult > 0 else 0
   639	        if rr < 1.3:
   640	            errors.append(
   641	                f"[{self.profile_name}] ATR R/R terlalu rendah: "
   642	                f"{self.atr_tp_mult}/{self.atr_sl_mult} = {rr:.2f} (minimum 1.3)"
   643	            )
   644	
   645	        if self.volume_spike <= self.volume_mult:
   646	            errors.append(
   647	                f"[{self.profile_name}] volume_spike ({self.volume_spike}) "
   648	                f"<= volume_mult ({self.volume_mult})"
   649	            )
   650	
   651	        if self.trailing_act_pct <= self.trailing_gap_pct:
   652	            errors.append(
   653	                f"[{self.profile_name}] trailing_act_pct ({self.trailing_act_pct}) "
   654	                f"<= trailing_gap_pct ({self.trailing_gap_pct})"
   655	            )
   656	
   657	        if not (40.0 <= self.entry_threshold <= 95.0):
   658	            errors.append(
   659	                f"[{self.profile_name}] entry_threshold {self.entry_threshold} "
   660	                f"di luar [40, 95]"
   661	            )
   662	
   663	        if not (0.0 <= self.confirmation_weight <= 1.0):
   664	            errors.append(
   665	                f"[{self.profile_name}] confirmation_weight {self.confirmation_weight} "
   666	                f"di luar [0, 1]"
   667	            )
   668	
   669	        return errors

   752	def _build_parameter_bounds(profile_name: str) -> ParameterBounds:
   753	    bounds  = ParameterBounds()
   754	    overrides = PARAMETER_BOUNDS_OVERRIDE.get(profile_name, {})
   755	
   756	    _BOUNDS_FIELD_MAP = {
   757	        "entry_threshold":  ("entry_threshold_min",  "entry_threshold_max"),
   758	        "volume_multiplier": ("volume_mult_min",      "volume_mult_max"),
   759	        "rsi_min":          ("rsi_min_lower",         "rsi_min_upper"),
   760	        "rsi_max":          ("rsi_max_lower",         "rsi_max_upper"),
   761	        "atr_sl_mult":      ("atr_sl_mult_min",       "atr_sl_mult_max"),
   762	        "atr_tp_mult":      ("atr_tp_mult_min",       "atr_tp_mult_max"),
   763	        "trailing_act_pct": ("trailing_act_pct_min",  "trailing_act_pct_max"),
   764	        "trailing_gap_pct": ("trailing_gap_pct_min",  "trailing_gap_pct_max"),
   765	        "quick_sl_pct":     ("quick_sl_pct_min",      "quick_sl_pct_max"),
   766	        "quick_tp_pct":     ("quick_tp_pct_min",      "quick_tp_pct_max"),
   767	    }
   768	
   769	    for param_key, (lo, hi) in overrides.items():
   770	        field_pair = _BOUNDS_FIELD_MAP.get(param_key)
   771	        if field_pair is None:
   772	            log.warning(
   773	                "PARAMETER_BOUNDS_OVERRIDE[%s][%s]: tidak ada mapping — diabaikan.",
   774	                profile_name, param_key,
   775	            )
   776	            continue
   777	        setattr(bounds, field_pair[0], lo)
   778	        setattr(bounds, field_pair[1], hi)
   779	
   780	        if lo >= hi:
   781	            raise ValueError(
   782	                f"PARAMETER_BOUNDS_OVERRIDE[{profile_name}][{param_key}]: "
   783	                f"min {lo} >= max {hi}. Ini adalah konfigurasi yang tidak valid."
   784	            )
   785	
   786	    return bounds

   873	def compare_profiles(
   874	    profile_a: str,
   875	    profile_b: str,
   876	    fields: Optional[List[str]] = None,
   877	) -> Dict[str, Dict[str, Any]]:
   878	    thresh_a = get_profile_thresholds(profile_a)
   879	    thresh_b = get_profile_thresholds(profile_b)
   880	
   881	    _NUMERIC_FIELDS = [
   882	        "entry_threshold", "volume_mult", "volume_spike",
   883	        "rsi_min", "rsi_max", "rsi_gc_min", "min_breakout_pct",
   884	        "atr_sl_mult", "atr_tp_mult", "atr_pct_threshold",
   885	        "quick_sl_pct", "quick_tp_pct",
   886	        "trailing_act_pct", "trailing_gap_pct",
   887	        "typical_atr_pct", "max_spread_pct",
   888	        "confirmation_weight", "confirmation_min_score",
   889	    ]
   890	
   891	    target_fields = fields if fields else _NUMERIC_FIELDS
   892	    result: Dict[str, Dict[str, Any]] = {}
   893	
   894	    for f in target_fields:
   895	        val_a = getattr(thresh_a, f, None)
   896	        val_b = getattr(thresh_b, f, None)
   897	        if val_a is None or val_b is None:
   898	            continue
   899	        try:
   900	            diff = float(val_b) - float(val_a)
   901	        except (TypeError, ValueError):
   902	            diff = None
   903	        result[f] = {"a": val_a, "b": val_b, "diff": diff}
   904	
   905	    return result

    56	        _MAP: Dict[str, Tuple[str, str]] = {

    78	        bounds = self.get_bounds(param_name)

    81	        try:
    82	            v = float(value)

   327	        cur_atr_pct = (cur_atr / cur_price * 100) if cur_price > 0 else 0.0
   328	        typical_atr = profile.typical_atr_pct or 1.0
   329	        atr_ratio = cur_atr_pct / typical_atr if typical_atr > 0 else 1.0

   333	            tp_mult = profile.atr_tp_mult * 1.2
   334	            mode = "HIGH_VOLATILITY"
   335	        elif atr_ratio > 1.5:

   337	            tp_mult = profile.atr_tp_mult * 1.1
   338	            mode = "ELEVATED_VOLATILITY"
   339	        elif atr_ratio < 0.5:
   340	            sl_mult = profile.atr_sl_mult * 0.9
   341	            tp_mult = profile.atr_tp_mult * 0.9
   342	            mode = "LOW_VOLATILITY"
   343	        else:
   344	            sl_mult = profile.atr_sl_mult
   345	            tp_mult = profile.atr_tp_mult
   346	            mode = "NORMAL"

   348	        vol_threshold = profile.volume_mult * (0.85 if cur_vol_ratio > 3.0 else 1.0)
   349	        if atr_ratio > 1.5:
   350	            rsi_min = max(profile.rsi_min - 5, 30)
   351	            rsi_max = min(profile.rsi_max + 5, 85)

   354	            rsi_max = profile.rsi_max

   356	        sl_atr = cur_atr * sl_mult
   357	        sl_pct = cur_price * (profile.quick_sl_pct / 100)
   358	        sl_final = cur_price - max(sl_atr, sl_pct)
   359	
   360	        tp_atr = cur_atr * tp_mult
   361	        tp_pct = cur_price * (profile.quick_tp_pct / 100)
   362	        tp_final = cur_price + max(tp_atr, tp_pct)
   363	
   364	        return {
   365	            "profile": profile.profile.value,
   366	            "adaptive_mode": mode,
   367	            "atr_ratio": round(atr_ratio, 2),
   368	            "atr_pct": round(cur_atr_pct, 4),
   369	            "sl_mult": round(sl_mult, 2),
   370	            "tp_mult": round(tp_mult, 2),
   371	            "sl": round(sl_final, 8),
   372	            "tp": round(tp_final, 8),
   373	            "vol_threshold": round(vol_threshold, 2),
   374	            "rsi_min": rsi_min,
   375	            "rsi_max": rsi_max,
   376	            "timeframe": profile.timeframe,
   377	            "trailing_act_pct": profile.trailing_act_pct,
   378	            "trailing_gap_pct": profile.trailing_gap_pct,
   379	        }

    29	def _fire_and_forget_db(coro_or_result) -> None:
    30	    """
    31	    [TAMBAHAN] Helper untuk handle db calls yang bisa sync ATAU async.
    32	    registry.py dipanggil dari konteks sync (set_profile_override,
    33	    revert_parameter_override, dll) tapi DatabaseManager menggunakan
    34	    async methods. Tanpa penanganan ini, coroutine dari async DB call
    35	    tidak pernah dieksekusi — perubahan tidak tersimpan ke DB secara
    36	    diam-diam tanpa error.
    37	
    38	    Solusi: kalau ada running event loop, schedule via create_task.
    39	    Kalau tidak ada running loop (konteks sync murni), tutup coroutine
    40	    agar tidak memory leak dan log debug.
    41	    Konsisten dengan cara apply_parameter_override sudah melakukannya.
    42	    """
    43	    if coro_or_result is None:
    44	        return
    45	    if not hasattr(coro_or_result, "__await__"):
    46	        return  # sudah sync result
    47	    try:
    48	        loop = asyncio.get_running_loop()
    49	        loop.create_task(coro_or_result)
    50	    except RuntimeError:
    51	        try:
    52	            coro_or_result.close()
    53	        except Exception:
    54	            pass
    55	        log.debug(
    56	            "DB call tidak bisa di-schedule (no running loop) — "
    57	            "perubahan mungkin tidak tersimpan ke DB."
    58	        )

   109	def _build_coin_profile(symbol: str, profile_name: str, overrides: Optional[Dict[str, Any]] = None) -> CoinProfile:
   110	    thresh = get_profile_thresholds(profile_name)
   111	    kwargs = thresh.to_coinprofile_kwargs()
   112	
   113	    timeframe = PROFILE_TIMEFRAME.get(profile_name, "1h")
   114	    timeframe_conf = TIMEFRAME_CONFIRMATION_MAP.get(timeframe, "4h")
   115	
   116	    if overrides:
   117	        for param, value in overrides.items():
   118	            if param in kwargs:
   119	                if thresh.param_bounds.is_within_bounds(param, value):
   120	                    kwargs[param] = value
   121	                    log.info(
   122	                        "Applied override: %s/%s | %s = %s",
   123	                        symbol, profile_name, param, value,
   124	                    )
   125	                else:
   126	                    bounds = thresh.param_bounds.get_bounds(param)
   127	                    log.warning(
   128	                        "Override '%s' = %s di luar bounds %s untuk %s/%s — diabaikan.",
   129	                        param, value, bounds, symbol, profile_name,
   130	                    )
   131	
   132	    profile_enum = StrategyProfile(profile_name)
   133	
   134	    coin_profile = CoinProfile(
   135	        symbol=symbol,
   136	        profile=profile_enum,
   137	        timeframe=timeframe,
   138	        timeframe_conf=timeframe_conf,
   139	        description=f"{PROFILE_EMOJI.get(profile_name, '⚙️')} {profile_name} | {symbol}",
   140	        **kwargs,
   141	    )
   142	
   143	    errors = coin_profile.validate()
   144	    if errors:
   145	        log.error(
   146	            "CoinProfile %s/%s validation errors: %s",
   147	            symbol, profile_name, errors,
   148	        )
   149	
   150	    return coin_profile

   198	def select_profile_from_indicators(
   199	    symbol: str,
   200	    ind_volatility: float = 50.0,
   201	    ind_momentum:   float = 50.0,
   202	    ind_trend:      float = 50.0,
   203	    ema_stack_score: float = 50.0,
   204	    adx:            float = 20.0,
   205	    rsi:            float = 50.0,
   206	    atr_pct:        float = 0.5,
   207	    regime:         str   = "trending_bull",
   208	) -> str:
   209	    """
   210	    Pilih profile otomatis berdasarkan kondisi indikator saat itu.
   211	    Koin tetap punya jati diri (base profile dari _COIN_PROFILE_MAP)
   212	    tapi bisa naik/turun profile sesuai kondisi lapangan.
   213	    """
   214	    base = symbol.split("/")[0].upper()
   215	    base_profile = _COIN_PROFILE_MAP.get(base, "scalp_volatile")
   216	
   217	    # Kondisi extreme_momentum — semua koin bisa masuk kalau syarat terpenuhi
   218	    if (
   219	        ema_stack_score >= 88
   220	        and ind_trend    >= 80
   221	        and ind_momentum >= 78
   222	        and adx          >= 30
   223	        and rsi          >= 60
   224	    ):
   225	        return "extreme_momentum"
   226	
   227	    # Kondisi trend_follow — tren kuat, volatilitas normal
   228	    if (
   229	        ind_trend    >= 75
   230	        and ema_stack_score >= 75
   231	        and adx          >= 25
   232	        and ind_volatility <= 70
   233	        and regime == "trending_bull"
   234	    ):
   235	        return "trend_follow"
   236	
   237	    # Kondisi breakout_swift — momentum tinggi + volatilitas sedang
   238	    if (
   239	        ind_momentum >= 70
   240	        and ind_trend >= 65
   241	        and adx       >= 22
   242	        and atr_pct   >= 0.4
   243	    ):
   244	        return "breakout_swift"
   245	
   246	    # Kondisi scalp_volatile — volatilitas tinggi
   247	    if (
   248	        ind_volatility >= 72
   249	        or atr_pct      >= 1.2
   250	        or regime == "volatile_expansion"
   251	    ):
   252	        return "scalp_volatile"
   253	
   254	    # Kondisi mean_revert — pasar ranging, momentum lemah
   255	    if (
   256	        regime == "ranging"
   257	        or (adx <= 20 and ind_momentum <= 45)
   258	    ):
   259	        return "mean_revert"
   260	
   261	    # Kondisi hodl_accumulate — tren lemah, semua indikator netral
   262	    if (
   263	        ind_trend    <= 55
   264	        and ind_momentum <= 50
   265	        and adx          <= 20
   266	    ):
   267	        return "hodl_accumulate"
   268	
   269	    # Fallback — pakai jati diri koin
   270	    return base_profile

   275	    db_manager=None,
   276	) -> CoinProfile:
   277	
   278	    base = symbol.split("/")[0].upper()

   286	    if override_profile:

   297	        profile_name = _get_default_profile_name(base)
   298	
   299	    param_overrides = _ACTIVE_OVERRIDES.get(base, {})
   300	
   301	    if profile_name is None:

   325	def auto_classify_profile(base: str, ticker: dict, spread_pct: float = 0.0) -> str:
   326	    """
   327	    Klasifikasi otomatis profil koin berdasarkan data live ticker.
   328	    Dipakai sebagai fallback kalau koin tidak ada di _COIN_PROFILE_MAP.
   329	
   330	    Parameter:
   331	        base       : simbol base coin (misal "SOL", "BONK")
   332	        ticker     : dict dari ws_feed.live_tickers[symbol]
   333	        spread_pct : spread % dari ws_feed.get_spread_pct(symbol)
   334	
   335	    Return:
   336	        nama profil (str) — salah satu dari StrategyProfile values
   337	    """
   338	    try:
   339	        quote_volume = float(ticker.get("quote_volume") or 0)
   340	        last         = float(ticker.get("last") or 0)
   341	        # Fallback volume dari base_volume * price
   342	        if quote_volume <= 0 and last > 0:
   343	            quote_volume = float(ticker.get("volume") or 0) * last
   344	    except Exception:
   345	        quote_volume = 0.0
   346	
   347	    spread = abs(spread_pct) if spread_pct else 0.0
   348	
   349	    # ── Klasifikasi berdasarkan volume + spread ──────────────────
   350	    # Tier 1: Blue chip — volume sangat tinggi, spread sangat kecil
   351	    if quote_volume >= 50_000_000 and spread < 0.15:
   352	        profile = "hodl_accumulate"
   353	
   354	    # Tier 2: Mid cap stabil — volume tinggi, spread kecil
   355	    elif quote_volume >= 10_000_000 and spread < 0.25:
   356	        profile = "trend_follow"
   357	
   358	    # Tier 3: Liquid aktif — volume cukup, spread wajar
   359	    elif quote_volume >= 3_000_000 and spread < 0.40:
   360	        profile = "breakout_swift"
   361	
   362	    # Tier 4: Volatile liquid — volume cukup tapi spread lebar
   363	    elif quote_volume >= 1_000_000 and spread < 0.80:
   364	        profile = "scalp_volatile"
   365	
   366	    # Tier 5: Meme/pump — volume cukup tapi spread sangat lebar
   367	    elif quote_volume >= 500_000 and spread >= 0.80:
   368	        profile = "extreme_momentum"
   369	
   370	    # Tier 6: Low volume — cenderung ranging, mean revert
   371	    elif quote_volume >= 100_000:
   372	        profile = "mean_revert"
   373	
   374	    # Tier 7: Sangat low volume — pakai mean_revert (paling jarang trade, paling aman)
   375	    else:
   376	        profile = "mean_revert"
   377	
   378	    # Cache hasil ke _COIN_PROFILE_MAP runtime (tidak ubah file)
   379	    _COIN_PROFILE_MAP[base] = profile
   380	    log.info(
   381	        "AutoClassify: %s → %s | vol_24h=$%.0f spread=%.3f%%",
   382	        base, profile, quote_volume, spread,
   383	    )
   384	    return profile

   386	def _resolve_profile_name(name: str) -> Optional[str]:
   387	    if not name:
   388	        return None
   389	    lower = name.lower()
   390	
   391	    known = get_all_profile_names()
   392	    if lower in known:
   393	        return lower
   394	
   395	    if lower in _LEGACY_PROFILE_ALIAS:
   396	        return _LEGACY_PROFILE_ALIAS[lower]
   397	
   398	    for prof in StrategyProfile:
   399	        if prof.value == lower:
   400	            return prof.value
   401	
   402	    return None

   404	def set_profile_override(
   405	    symbol: str,
   406	    profile_name: str,
   407	    db_manager=None,
   408	) -> Tuple[bool, str]:
   409	    resolved = _resolve_profile_name(profile_name)
   410	    if resolved is None:
   411	        valid = get_all_profile_names() + list(_LEGACY_PROFILE_ALIAS.keys())
   412	        return False, (
   413	            f"Profile '{profile_name}' tidak dikenal. "
   414	            f"Valid: {sorted(set(valid))}"
   415	        )
   416	
   417	    base = symbol.split("/")[0].upper()
   418	    _MANUAL_PROFILE_OVERRIDES[base] = resolved
   419	    _invalidate_cache(base)
   420	
   421	    if db_manager is not None:
   422	        try:
   423	            # [BUG-FIX] Sebelumnya: save_parameter_change dipanggil tanpa
   424	            # penanganan async — DatabaseManager.save_parameter_change adalah
   425	            # async method, tanpa await/create_task coroutine tidak pernah
   426	            # dieksekusi, override tidak tersimpan ke DB secara diam-diam.
   427	            # Sekarang: pakai _fire_and_forget_db yang konsisten dengan cara
   428	            # apply_parameter_override sudah melakukannya.
   429	            _fire_and_forget_db(db_manager.save_parameter_change(
   430	                symbol=base,
   431	                profile=resolved,
   432	                parameter_name="_profile_override",
   433	                old_value=_COIN_PROFILE_MAP.get(base, "unknown"),
   434	                new_value=resolved,
   435	                reason="Manual profile override via set_profile_override()",
   436	                approved_by="manual",
   437	            ))
   438	        except Exception as exc:
   439	            log.warning("Gagal simpan profile override ke DB: %s", exc)
   440	
   441	    log.info("Profile override set: %s → %s", base, resolved)
   442	    profile = get_coin_profile(symbol)
   443	    return True, (
   444	        f"Override berhasil: {symbol} → {resolved} "
   445	        f"(TF={profile.timeframe}, threshold={profile.entry_threshold})"
   446	    )

   448	def clear_profile_override(symbol: str, db_manager=None) -> bool:
   449	    base = symbol.split("/")[0].upper()
   450	    existed = base in _MANUAL_PROFILE_OVERRIDES
   451	
   452	    if existed:
   453	        # [BUG-FIX] Sebelumnya old_value dibaca dari _MANUAL_PROFILE_OVERRIDES
   454	        # SETELAH `del _MANUAL_PROFILE_OVERRIDES[base]` -- karena key sudah
   455	        # dihapus, .get(base, "override") SELALU jatuh ke default "override",
   456	        # tidak pernah menampilkan profile asli yang di-clear. Dibuktikan via
   457	        # eksperimen: old_value yang tersimpan ke DB selalu literal "override"
   458	        # bukan "trend_follow"/dst. Dampak: audit-trail parameter_history di
   459	        # DB jadi tidak akurat (bukan bug yang mempengaruhi trading, cuma
   460	        # riwayat log). Fix: simpan nilai lama SEBELUM del.
   461	        old_profile_value = _MANUAL_PROFILE_OVERRIDES[base]
   462	        del _MANUAL_PROFILE_OVERRIDES[base]
   463	        _invalidate_cache(base)
   464	        log.info("Profile override cleared: %s", base)
   465	
   466	        if db_manager is not None:
   467	            try:
   468	                # [BUG-FIX] Sama seperti set_profile_override — async call
   469	                # tanpa penanganan yang benar. Fix: _fire_and_forget_db.
   470	                _fire_and_forget_db(db_manager.save_parameter_change(
   471	                    symbol=base,
   472	                    profile=_COIN_PROFILE_MAP.get(base, "unknown"),
   473	                    parameter_name="_profile_override",
   474	                    old_value=old_profile_value,
   475	                    new_value="(cleared)",
   476	                    reason="Profile override cleared",
   477	                    approved_by="manual",
   478	                ))
   479	            except Exception as exc:
   480	                log.warning("Gagal simpan clear override ke DB: %s", exc)
   481	
   482	    return existed

   484	def apply_parameter_override(
   485	    symbol: str,
   486	    profile_name: str,
   487	    parameter_name: str,
   488	    new_value: Any,
   489	    source: str = "meta_learner",
   490	    db_manager=None,
   491	) -> Tuple[bool, str]:
   492	    base = symbol.split("/")[0].upper()
   493	
   494	    profile_name_resolved = (
   495	        _MANUAL_PROFILE_OVERRIDES.get(base)
   496	        or _COIN_PROFILE_MAP.get(base)
   497	    )
   498	    if profile_name_resolved is None:
   499	        return False, f"Coin '{base}' tidak terdaftar, tidak bisa apply parameter override."
   500	
   501	    thresh = get_profile_thresholds(profile_name_resolved)
   502	    if not thresh.param_bounds.is_within_bounds(parameter_name, new_value):
   503	        bounds = thresh.param_bounds.get_bounds(parameter_name)
   504	        return False, (
   505	            f"Parameter '{parameter_name}' = {new_value} di luar bounds {bounds} "
   506	            f"untuk {base}/{profile_name_resolved}."
   507	        )
   508	
   509	    old_profile = get_coin_profile(symbol)
   510	    old_value = getattr(old_profile, parameter_name, None)
   511	
   512	    if base not in _ACTIVE_OVERRIDES:
   513	        _ACTIVE_OVERRIDES[base] = {}
   514	    _ACTIVE_OVERRIDES[base][parameter_name] = new_value
   515	    _invalidate_cache(base)
   516	
   517	    if db_manager is not None:
   518	        try:
   519	            res = db_manager.save_parameter_change(
   520	                symbol=base,
   521	                profile=profile_name_resolved,
   522	                parameter_name=parameter_name,
   523	                old_value=old_value,
   524	                new_value=new_value,
   525	                reason=f"Applied by {source}",
   526	                approved_by=source,
   527	            )
   528	            if hasattr(res, "__await__"):
   529	                try:
   530	                    asyncio.get_running_loop().create_task(res)
   531	                except RuntimeError:
   532	                    # No running loop; best-effort fire-and-forget.
   533	                    # [BUG-FIX minor] Sebelumnya coroutine `res` dibiarkan
   534	                    # begitu saja di sini kalau tidak ada running loop --
   535	                    # tidak di-close() spt _fire_and_forget_db (helper serupa
   536	                    # di file yg sama) melakukannya, sehingga Python akan
   537	                    # memunculkan RuntimeWarning "coroutine was never awaited"
   538	                    # saat garbage collection. Bukan bug fungsional (data
   539	                    # override tetap tersimpan ke _ACTIVE_OVERRIDES in-memory),
   540	                    # cuma noise di log/warning. Konsistenkan dgn pola
   541	                    # _fire_and_forget_db.
   542	                    try:
   543	                        res.close()
   544	                    except Exception:
   545	                        pass
   546	        except Exception as exc:
   547	            log.warning("Gagal simpan parameter override ke DB: %s", exc)
   548	
   549	    log.info(
   550	        "Parameter override applied: %s/%s | %s: %s → %s (source=%s)",
   551	        base, profile_name_resolved, parameter_name, old_value, new_value, source,
   552	    )
   553	    return True, (
   554	        f"Override applied: {base}/{profile_name_resolved} | "
   555	        f"{parameter_name}: {old_value} → {new_value}"
   556	    )

   599	async def _load_overrides_from_db_async(symbol: str, db_manager) -> None:
   600	    """
   601	    # [TAMBAHAN] Bagian async sebenarnya dari _load_overrides_from_db.
   602	    # Dipisah dari fungsi sync di bawah supaya bisa di-await langsung
   603	    # (dipanggil dari konteks async seperti startup main.py) ATAU
   604	    # di-fire-and-forget (dipanggil lazy dari get_coin_profile yang sync).
   605	    """
   606	    try:
   607	        history = await db_manager.get_parameter_history(symbol=symbol, limit=100)
   608	        if not history:
   609	            return
   610	
   611	        applied: Dict[str, Any] = {}
   612	        for record in sorted(history, key=lambda r: r.get("timestamp", ""), reverse=True):
   613	            param = record.get("parameter_name", "")
   614	            if param.startswith("_"):
   615	                continue
   616	            if param not in applied:
   617	                new_val = record.get("new_value")
   618	                if new_val is not None and new_val != "(reverted to default)":
   619	                    applied[param] = new_val
   620	
   621	        if applied:
   622	            _ACTIVE_OVERRIDES[symbol] = applied
   623	            log.debug(
   624	                "Loaded %d parameter overrides dari DB untuk %s: %s",
   625	                len(applied), symbol, list(applied.keys()),
   626	            )
   627	
   628	    except Exception as exc:
   629	        log.warning("Gagal load overrides dari DB untuk %s: %s", symbol, exc)
